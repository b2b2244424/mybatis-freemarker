29  public class CustomizedDataContextTest {
30      protected static SqlSessionFactory sqlSessionFactory;

32      @BeforeClass
33      public static void setUp() throws Exception {
34          Class.forName("org.hsqldb.jdbcDriver");
35  
36          JDBCDataSource dataSource = new JDBCDataSource();
37          dataSource.setUrl("jdbc:hsqldb:mem:db4");
38          dataSource.setUser("sa");
39          dataSource.setPassword("");
40  
41          try (Connection conn = dataSource.getConnection()) {
42              try (Reader reader = Resources.getResourceAsReader("org/mybatis/scripting/freemarker/create-db.sql")) {
43                  ScriptRunner runner = new ScriptRunner(conn);
44                  runner.setLogWriter(null);
45                  runner.setErrorLogWriter(null);
46                  runner.runScript(reader);
47                  conn.commit();
48              }
49          }
50  
51          TransactionFactory transactionFactory = new JdbcTransactionFactory();
52          Environment environment = new Environment("development", transactionFactory, dataSource);
53  
54          // You can call configuration.setDefaultScriptingLanguage(FreeMarkerLanguageDriver.class)
55          // after this to use FreeMarker driver by default.
56          Configuration configuration = new Configuration(environment);
57  
58          configuration.addMapper(CustomizedDataContextMapper.class);
59          sqlSessionFactory = new SqlSessionFactoryBuilder().build(configuration);
60      }
61  
62      public static class CustomSqlSource extends FreeMarkerSqlSource {
63          public CustomSqlSource(Template template, Configuration configuration) {
64              super(template, configuration);
65          }
66  
67          @Override
68          protected Object preProcessDataContext(Object dataContext, boolean isMap) {
69              dataContext = super.preProcessDataContext(dataContext, isMap);
70              if (isMap) {
71                  ((Map<String, Object>) dataContext).put("MY_NAME", new SimpleScalar("Barney"));
72              } else {
73                  ((ParamObjectAdapter) dataContext).putAdditionalParam("MY_NAME", new SimpleScalar("Barney"));
74              }
75              return dataContext;
76          }
77      }
78  
79      public static class CustomFreeMarkerLanguageDriver extends FreeMarkerLanguageDriver {
80          @Override
81          protected SqlSource createSqlSource(Template template, Configuration configuration) {
82              return new CustomSqlSource(template, configuration);
83          }
84      }
85  
86      @Test
87      public void test() {
88          try (SqlSession sqlSession = sqlSessionFactory.openSession()) {
89              CustomizedDataContextMapper mapper = sqlSession.getMapper(CustomizedDataContextMapper.class);
90              List<Name> names = mapper.find();
91              Assert.assertTrue(names.size() == 1);
92          }
93      }
